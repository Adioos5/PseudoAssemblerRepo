TAB 	   DS  5*INTEGER
ILE        DC  INTEGER(5)
CZTERY     DC  INTEGER(4)
ZERO       DC  INTEGER(0)
JEDEN      DC  INTEGER(1)
WYNIK      DS  INTEGER

	   LA 1, TAB
	   SR 2, 2
	   SR 5, 5
PETLA	   L  3, 0(1)
	   C  3, ZERO
	   JN PRZESUN
	   A  5, 3
PRZESUN	   A  1, CZTERY
           A  2, JEDEN
	   C  2, ILE
	   JZ KONIEC
	   J  PETLA
KONIEC     ST 5, WYNIK
